// Saving and rendering tasks with localStorage persistence
function saveTasks(tasks) {
  localStorage.setItem('focusos-tasks', JSON.stringify(tasks));
}

function renderTasks(tasks) {
  list.innerHTML = '';
  tasks.forEach((task, index) => {
    // Creates each task item with delete button
    const li = document.createElement('li');
    li.className = "bg-gray-800 p-2 rounded flex justify-between items-center";
    li.innerHTML = `
      <span>${task}</span>
      <button data-index="${index}" class="text-red-500 hover:text-red-700">🗑️</button>
    `;
    list.appendChild(li);
  });
}
